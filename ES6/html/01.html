生成器函数: 

            function* fibs() {
                let a = 0;
                let b = 1;
                while (true) {
                  yield a;
                  [a, b] = [b, a + b];
                }
              }
              
              var [first, second, third, fourth, fifth, sixth] = fibs();
              console.log(first+" "+second+" " + third+" "+fourth + " " + fifth + " "+ sixth);
              //0 1 1 2 3 5
