Symbol 对象属性的唯一标识符
        //Object.assign();
        var target  = {a : 1}; //目标对象
        var source1 = {b : 2}; //源对象1
        var source2 = {c : 3}; //源对象2
        var source3 = {c : 4}; //源对象3，和source2中的对象有同名属性c
        Object.assign(target,source1,source2,source3);
        //结果如下：
        //{a:1,b:2,c:4}
    
            var smy = Symbol();
            var info = {
                smy: 'x',
                [smy]: 'y'
            };

            console.log(info.smy);       // 输出'x'
            console.log(info['smy']);    // 输出'x'
            console.log(info[smy]);      // 输出'y'
